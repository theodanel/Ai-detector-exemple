const imageInput = document.getElementById("imageInput"); const imagePreview = document.getElementById("imagePreview"); const analyzeButton = document.getElementById("analyzeButton"); const resultsContent = document.getElementById("resultsContent"); const loadingMessage = document.getElementById("loadingMessage"); imageInput.addEventListener("change", () => { const file = imageInput.files[0]; if (!file) { return; } const imageUrl = URL.createObjectURL(file); imagePreview.src = imageUrl; imagePreview.style.display = "block"; }); analyzeButton.addEventListener("click", async () => { const file = imageInput.files[0]; if (!file) { alert("Sélectionnez une image."); return; } loadingMessage.classList.remove("hidden"); resultsContent.innerHTML = ""; const formData = new FormData(); formData.append("file", file); try { const response = await fetch( "http://127.0.0.1:8000/api/full-analyze-image", { method: "POST", body: formData } ); const data = await response.json(); displayResults(data); } catch (error) { resultsContent.innerHTML = `

Erreur pendant l'analyse.

`; console.error(error); } finally { loadingMessage.classList.add("hidden"); } }); function displayResults(data) { const analysis = data.analysis; const predictionsHtml = analysis.predictions .map(prediction => { const percent = Math.round( prediction.confidence * 100 ); return `
${prediction.aliment} ${percent}%
`; }) .join(""); const recommendationsHtml = data.recommendation.recommendations .map(recommendation => `
  • ${recommendation}
    • `) .join(""); resultsContent.innerHTML = `

    Fiabilité de l'analyse

    ${analysis.message}

    Confiance globale : ${Math.round(analysis.global_confidence * 100)}%

    Aliments détectés

    ${predictionsHtml}

    Macros nutritionnelles

    Calories
    ${Math.round(data.nutrition.total_calories)}
    Protéines
    ${data.nutrition.total_proteines} g
    Glucides
    ${data.nutrition.total_glucides} g
    Lipides
    ${data.nutrition.total_lipides} g

    Recommandations

      ${recommendationsHtml}
    `; }
